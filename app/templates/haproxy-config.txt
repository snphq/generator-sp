global
  maxconn 4096
  pidfile ~/tmp/haproxy-queue.pid

defaults
  log global
  log 127.0.0.1 local0
  log 127.0.0.1 local1 notice
  mode http
  timeout connect 300000
  timeout client 300000
  timeout server 300000
  maxconn 2000
  option redispatch
  retries 3
  option httpclose
  option httplog
  option forwardfor


frontend incoming
  bind *:80

  acl is_backend path_reg ^\/(api|auth|admin|assets|system|rich).*$
  acl is_frontend path_reg ^\/(bower_components|images|files|scripts|styles|favicon\.ico|robots\.txt|livereload\.js).*$

  use_backend www_api_backend  if is_backend
  use_backend www_frontend     if is_frontend
  default_backend www_frontend_pushstate

backend www_frontend
  server nodehost 127.0.0.1:9000

backend www_frontend_pushstate
  reqrep ^GET\ /(.*)\ HTTP/1.1 GET\ /index.html\ HTTP/1.1
  server nodehost 127.0.0.1:9000

backend www_api_backend
  reqirep ^Host: Host:\ hostname.t.snpdev.ru
  server nodehost hostname-2.t.snpdev.ru
